%% \hminwidth[*]{«横幅»}{«テキスト»}
% テキストを指定の横幅に"イイカンジに"出力する
% *-> 均等割り付けしない
\NewDocumentCommand\hminwidth{s m m}{{%
	\settowidth{\OrigWidth}{#3}% \settowidth 便利
	\ifthenelse{\lengthtest{\OrigWidth > #2}}{%
		% 指定幅より大きい場合は横方向に縮小する
		\resizebox{#2}{\height}{#3}% 横幅だけ変える
	}{%else
		\IfBooleanTF{#1}{%
			% * 付きで指定幅以下ならそのまま出力
			#3
		}{%
			% 指定幅以下の場合は均等割りを行う
			\kintou{#2}{\jghostguarded{}#3\jghostguarded{}}%
		}%
	}%
}}
